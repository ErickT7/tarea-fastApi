# 1. Usar una imagen oficial de Python ligera como base
FROM python:3.10-slim

# 2. Configurar variables de entorno para evitar que Python escriba archivos .pyc o use buffer
ENV PYTHONDONTWRITEBYTECODE=1
ENV PYTHONUNBUFFERED=1

# 3. Establecer el directorio de trabajo dentro del contenedor
WORKDIR /app

# 4. Instalar las dependencias del sistema necesarias para compilar paquetes si fuera necesario
RUN apt-get update && apt-get install -y --no-install-recommends gcc libpq-dev && rm -rf /var/lib/apt/lists/*

# 5. Instalar las dependencias de Python directamente
RUN pip install --no-cache-dir fastapi "uvicorn[standard]" sqlalchemy

# 6. Copiar todo el código de nuestra API dentro del contenedor
COPY . /app

# 7. Exponer el puerto 8000 que es donde escucha FastAPI
EXPOSE 8000

# 8. Comando para arrancar Uvicorn escuchando en todas las interfaces de red (0.0.0.0)
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
